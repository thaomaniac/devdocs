# Docusaurus

> React-based, rich ecosystem, good for large projects with versioning needs.

## Setup

```bash
npx create-docusaurus@latest my-docs classic --typescript
cd my-docs
npm install
npm run start
```

## docusaurus.config.ts

```ts
import type { Config } from '@docusaurus/types'
import type * as Preset from '@docusaurus/preset-classic'

const config: Config = {
  title: 'DevDocs',
  tagline: 'Personal developer reference',
  url: 'https://your-org.github.io',
  baseUrl: '/',
  organizationName: 'your-org',
  projectName: 'devdocs',
  themeConfig: {
    navbar: {
      title: 'DevDocs',
      items: [
        { type: 'docSidebar', sidebarId: 'mysqlSidebar', label: 'MySQL' }
      ]
    },
    prism: {
      additionalLanguages: ['bash', 'sql', 'php', 'yaml']
    }
  } satisfies Preset.ThemeConfig,
  presets: [
    [
      'classic',
      {
        docs: {
          routeBasePath: '/',
          sidebarPath: './sidebars.ts'
        },
        blog: false
      } satisfies Preset.Options
    ]
  ]
}

export default config
```

## sidebars.ts

**Option 1 — auto-generated** (order controlled via frontmatter/category files):

```ts
import type { SidebarsConfig } from '@docusaurus/plugin-content-docs'

const sidebars: SidebarsConfig = {
  mysqlSidebar: [{ type: 'autogenerated', dirName: '.' }]
}

export default sidebars
```

Control order with `sidebar_position` in each file's frontmatter:

```md
---
sidebar_position: 1
sidebar_label: Overview
---
```

Control folder label and order with `_category_.json` inside each folder:

```json
{
  "label": "Admin",
  "position": 3
}
```

**Option 2 — explicit** (full control, must update manually when adding files):

```ts
import type { SidebarsConfig } from '@docusaurus/plugin-content-docs'

const sidebars: SidebarsConfig = {
  mysqlSidebar: [
    { type: 'doc', id: 'mysql/index', label: 'Overview' },
    {
      type: 'category',
      label: 'Backup',
      items: ['mysql/backup/mysqldump']
    },
    {
      type: 'category',
      label: 'Admin',
      items: [
        'mysql/admin/performance',
        'mysql/admin/drop-tables'
      ]
    },
    { type: 'doc', id: 'mysql/magento', label: 'Magento' },
    {
      type: 'category',
      label: 'Reference',
      items: ['mysql/reference/cheatsheet']
    }
  ]
}

export default sidebars
```

## Note: syntax highlighting

After adding languages to `prism.additionalLanguages`, **restart the dev server** — hot reload does not apply this change.

## .github/workflows/docs.yml

```yaml
name: Deploy Docs
on:
  push:
    branches: [main]
permissions:
  contents: read
  pages: write
  id-token: write
jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
      - run: npm ci
      - run: npm run build
      - uses: actions/upload-pages-artifact@v3
        with:
          path: build
      - uses: actions/deploy-pages@v4
```
